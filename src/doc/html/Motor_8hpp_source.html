156 void drive(const Direction motorDirection, const int specifiedSpeed = 0) {
157 const int driveSpeed = specifiedSpeed > 0 ? specifiedSpeed : speed;
158
159 motorPinWrite(r_EN_Pin, HIGH);
160 motorPinWrite(l_EN_Pin, HIGH);
161
162 switch (motorDirection) {
163 case Direction::EXTEND:
164 ledcWrite(pwmRChannel, driveSpeed);
165 ledcWrite(pwmLChannel, 0);
166 break;
167 case Direction::STOP:
168 ledcWrite(pwmRChannel, 0);
169 ledcWrite(pwmLChannel, 0);
170 motorPinWrite(r_EN_Pin, LOW);
171 motorPinWrite(l_EN_Pin, LOW);
172 break;
173 case Direction::RETRACT:
174 ledcWrite(pwmRChannel, 0);
175 ledcWrite(pwmLChannel, driveSpeed);
176 break;
177 default:
178 break;
179 } // end direction handler
180
181 lastPos = pos;
182 READ_POSITION_ENCODER()
183 } // end drive
