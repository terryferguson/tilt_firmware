157 void drive(const Direction motorDirection, const int specifiedSpeed = 0) {
158 const int driveSpeed = specifiedSpeed > 0 ? specifiedSpeed : speed;
159
160 switch (motorDirection) {
161 case Direction::EXTEND:
162 motorPinWrite(r_EN_Pin, HIGH);
163 motorPinWrite(l_EN_Pin, HIGH);
164 ledcWrite(pwmRChannel, driveSpeed);
165 ledcWrite(pwmLChannel, 0);
166 break;
167 case Direction::STOP:
168 motorPinWrite(r_EN_Pin, HIGH);
169 motorPinWrite(l_EN_Pin, HIGH);
170 ledcWrite(pwmRChannel, 0);
171 ledcWrite(pwmLChannel, 0);
172 motorPinWrite(r_EN_Pin, LOW);
173 motorPinWrite(l_EN_Pin, LOW);
174 break;
175 case Direction::RETRACT:
176 motorPinWrite(r_EN_Pin, HIGH);
177 motorPinWrite(l_EN_Pin, HIGH);
178 ledcWrite(pwmRChannel, 0);
179 ledcWrite(pwmLChannel, driveSpeed);
180 break;
181 default:
182 break;
183 } // end direction handler

185 lastPos = pos;
186 READ_POSITION_ENCODER()
187 } // end drive

286 void update(const int newSpeed = (MAX_SPEED + 1)) {
287 if (desiredPos >= 0) {
288 if (pos > desiredPos) {
289 dir = Direction::RETRACT;
290 } else if (pos < desiredPos) {
291 dir = Direction::EXTEND;
292 } else {
293 dir = Direction::STOP;
294 desiredPos = -1;
295 displayInfo();
296 }
297 }
298
299 // For lift column - Extension limit
300 if (dir == Direction::EXTEND && (pos) > totalPulseCount) {
301 dir = Direction::STOP;
302 return;
303 }
304
305 // For lift column - Retraction limit
306 if (dir == Direction::RETRACT && (pos) < 50) {
307 dir = Direction::STOP;
308 return;
309 }
310
311 if (newSpeed > MAX_SPEED || newSpeed < 0) {
312 drive(dir, this->speed);
313 } else {
314 drive(dir, newSpeed);
315 }
316 }
