262 void setSpeed(int newSpeed) {
263 targetSpeed = newSpeed;
264 softStart = lastPWMUpdate = micros();
265
266 // Calculate the difference between the current speed and the requested
267 // speed and divide that difference by the number of update steps to get
268 // the PWM duty cycle increase/decrease per step.
269 //
270 // This will usually have a fractional part, so we make it a float value. We
271 // handle the rounding and conversion to an integer in the update method.
272 pwmUpdateAmount =
273 ceil((float)abs(targetSpeed - speed) / SOFT_MOVEMENT_UPDATE_STEPS);
274
275 // If the new speed is lower, make it negative, as we add the
276 // pwmUpdateAmount to the speed
277 if (targetSpeed < speed) {
278 pwmUpdateAmount = -pwmUpdateAmount;
279 }
280
281 if (debugEnabled) {
282 Serial.printf("MotorController\n"
283 "------------\n"
284 "setSpeed(%d)\n"
285 "speed: %3d\n"
286 "target speed: %3d\n"
287 "pwmUpdateAmount: %3.6f\n\n",
288 newSpeed,
289 speed,
290 targetSpeed,
291 pwmUpdateAmount);
292 }

358 void update(const float deltaT = 0.0f) {
359 // Only update PID if we're not stopped
360 if (Direction::STOP != systemDirection) {
361 ALL_MOTORS_COMMAND(readPos)
362
363 if (Direction::EXTEND == systemDirection) {
364 if (motors[LEADER].getNormalizedPos() <
365 motors[FOLLOWER].getNormalizedPos()) {
366 laggingIndex = LEADER;
367 leadingIndex = FOLLOWER;
368 } else {
369 laggingIndex = FOLLOWER;
370 leadingIndex = LEADER;
371 }
372 } else {
373 if (motors[LEADER].getNormalizedPos() >
374 motors[FOLLOWER].getNormalizedPos()) {
375 laggingIndex = LEADER;
376 leadingIndex = FOLLOWER;
377 } else {
378 laggingIndex = FOLLOWER;
379 leadingIndex = LEADER;
380 }
381 }
382 }
383
384 // Difference between current speed and target speed
385 const int speedDelta = abs(speed - targetSpeed);
386
387 // Get current time
388 const int currentTime = micros();
389
390 // Time since soft move has started
391 const int moveTimeDelta = currentTime - softStart;
392
393 // Time since PWM duty cycle (speed) was last modified
394 const int updateTimeDelta = currentTime - lastPWMUpdate;
395
396 // If we have a target speed
397 if (targetSpeed >= 0) {
398 // Are we ready to update?
399 if (updateTimeDelta >= SOFT_MOVEMENT_PWM_UPDATE_INTERVAL_MICROS) {
400 // If so, first check if the distance to target is less than our step
401 // amount. Or if the time to ramp up has expired
402 if (speedDelta < abs(pwmUpdateAmount) ||
403 moveTimeDelta >= SOFT_MOVEMENT_MICROS) {
404 // If so, then set the current speed to target speed and reset state
405 speed = targetSpeed;
406 RESET_SOFT_MOVEMENT
407
408 // If the requested direction is to stop, the finally update the system
409 // direction to stop
410 if (requestedDirection == Direction::STOP) {
411 systemDirection = Direction::STOP;
412 if (debugEnabled) {
413 Serial.println("System Direction: STOP");
414 const double timeSinceSoftStart = (double) (micros() - softStart) / (double) MICROS_IN_MS;
415
416 Serial.printf("Soft Movement PWM Update - "
417 "speed: %d - "
418 "target speed: %d - "
419 "time since soft start: %f ms - "
420 "pwmUpdateAmount: %f\n",
421 speed,
422 targetSpeed,
423 timeSinceSoftStart,
424 pwmUpdateAmount);
425 }
426
427 // Tell all motors to stop
428 immediateHalt();
429 }
430 }
431
432 // Otherwise, add the PWM update ammount to the current duty cycle
433 // (speed) Round the result and then convert to an integer
434 const float newSpeed = (float)speed + pwmUpdateAmount;
435 speed = (int)floorf(newSpeed);
436
437 // We just updated, so update last update timestamp
438 lastPWMUpdate = micros();
439
440 const double timeSinceSoftStart = (double) (micros() - softStart) / (double) MICROS_IN_MS;
441
442 if (debugEnabled) {
443 Serial.printf("Soft Movement PWM Update - "
444 "speed: %d - "
445 "target speed: %d - "
446 "time since soft start: %f ms - "
447 "pwmUpdateAmount: %f\n",
448 speed,
449 targetSpeed,
450 timeSinceSoftStart,
451 pwmUpdateAmount);
452 }
453 }
454 }
455
456 /*
457int maxCurrent = max(motors[leadingIndex].getCurrent(),
458 motors[laggingIndex].getCurrent());
459*/
460
461 // The error is the difference between the normalization of positions of
462 // the two motors
463 const float error = abs(motors[laggingIndex].getNormalizedPos() -
464 motors[leadingIndex].getNormalizedPos());
465
466 // Add the error from this timestep to the integral term
467 eIntegral += error * deltaT;
468
469 // Calculate the adjusted speed to set to the faster motor
470 //const int adjustedSpeed = speed - int((error * K_p) + (K_i * eIntegral));
471 const int adjustedSpeed = speed - int((error * K_p));
472
473 // Slow down the faster motor, keeping the value in the acceptable range
474 motors[leadingIndex].speed =
475 constrain(adjustedSpeed, 0, 2 << pwmResolution);
476 motors[laggingIndex].speed = speed;
477
478 // Update the motors with the adjustments
479 ALL_MOTORS_COMMAND(update)
