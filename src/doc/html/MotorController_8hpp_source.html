246void setSpeed(int newSpeed) {
247 targetSpeed = newSpeed;
248 softStart = lastPWMUpdate = micros();
249
250 // Calculate the difference between the current speed and the requested
251 // speed and divide that difference by the number of update steps to get
252 // the PWM duty cycle increase/decrease per step.
253 //
254 // This will usually have a fractional part, so we make it a float value. We
255 // handle the rounding and conversion to an integer in the update method.
256 pwmUpdateAmount =
257 ceil((float)abs(targetSpeed - speed) / SOFT_MOVEMENT_UPDATE_STEPS);
258
259 // If the new speed is lower, make it negative, as we add the
260 // pwmUpdateAmount to the speed
261 if (targetSpeed < speed) {
262 pwmUpdateAmount = -pwmUpdateAmount;
263 }
264
265 if (debugEnabled) {
266 Serial.printf("MotorController\n"
267 "------------\n"
268 "setSpeed(%d)\n"
269 "speed: %3d\n"
270 "target speed: %3d\n"
271 "pwmUpdateAmount: %3.6f\n\n",
272 newSpeed, speed, targetSpeed, pwmUpdateAmount);
273 }

340void update(const float deltaT = 0.0f) {
341 if (Direction::STOP == systemDirection) {
342 return;
343 }
344
345 ALL_MOTORS_COMMAND(readPos)
346
347 if (Direction::EXTEND == systemDirection) {
348 if (motors[LEADER].getNormalizedPos() <
349 motors[FOLLOWER].getNormalizedPos()) {
350 laggingIndex = MotorRoles::LEADER;
351 leadingIndex = MotorRoles::FOLLOWER;
352 } else {
353 laggingIndex = MotorRoles::FOLLOWER;
354 leadingIndex = MotorRoles::LEADER;
355 }
356 } else {
357 if (motors[LEADER].getNormalizedPos() >
358 motors[FOLLOWER].getNormalizedPos()) {
359 laggingIndex = MotorRoles::LEADER;
360 leadingIndex = MotorRoles::FOLLOWER;
361 } else {
362 laggingIndex = MotorRoles::FOLLOWER;
363 leadingIndex = MotorRoles::LEADER;
364 }
365 }
366
367 if (!pid_on) {
368 motors[leadingIndex].speed = 255;
369 motors[laggingIndex].speed = 255;
370 ALL_MOTORS_COMMAND(update);
371 return;
372 }
373
374 const int speedDelta = abs(speed - targetSpeed);
375 const int currentTime = micros();
376 const int moveTimeDelta = currentTime - softStart;
377 const int updateTimeDelta = currentTime - lastPWMUpdate;
378
379 if (targetSpeed < 0) {
380 return;
381 }
382
383 if (updateTimeDelta < SOFT_MOVEMENT_PWM_UPDATE_INTERVAL_MICROS) {
384 return;
385 }
386
387 if (speedDelta >= abs(pwmUpdateAmount) &&
388 moveTimeDelta < SOFT_MOVEMENT_MICROS) {
389 const float newSpeed = (float)speed + pwmUpdateAmount;
390 speed = (int)floorf(newSpeed);
391 lastPWMUpdate = micros();
392
393 if (!debugEnabled) {
394 return;
395 }
396
397 const double timeSinceSoftStart =
398 (double)(micros() - softStart) / (double)MICROS_IN_MS;
399
400 Serial.printf("Soft Movement PWM Update - "
401 "speed: %d - "
402 "target speed: %d - "
403 "time since soft start: %f ms - "
404 "pwmUpdateAmount: %f\n",
405 speed, targetSpeed, timeSinceSoftStart, pwmUpdateAmount);
406 } else {
407 speed = targetSpeed;
408 RESET_SOFT_MOVEMENT
409
410 if (requestedDirection == Direction::STOP) {
411 systemDirection = Direction::STOP;
412
413 if (debugEnabled) {
414 Serial.println("System Direction: STOP");
415 const double timeSinceSoftStart =
416 (double)(micros() - softStart) / (double)MICROS_IN_MS;
417
418 Serial.printf("Soft Movement PWM Update - "
419 "speed: %d - "
420 "target speed: %d - "
421 "time since soft start: %f ms - "
422 "pwmUpdateAmount: %f\n",
423 speed, targetSpeed, timeSinceSoftStart, pwmUpdateAmount);
424 }
425
426 immediateHalt();
427 }
428 }
429
430 const float error = abs(motors[laggingIndex].getNormalizedPos() -
431 motors[leadingIndex].getNormalizedPos());
432
433 eIntegral += error * deltaT;
434
435 const int adjustedSpeed = speed - int((error * K_p));
436
437 motors[leadingIndex].speed = constrain(adjustedSpeed, 0, 2 << pwmResolution);
438 motors[laggingIndex].speed = speed;
439
440 ALL_MOTORS_COMMAND(update);
441}
442}
