111 double integral;
112 double maxIntegral = 1000.0; // limit for integral wind-up
113 int intermediateSpeed = -1;

115 double control(double setpoint, double actualPosition) {
116 double error = setpoint - actualPosition;
117
118 integral += error;
119 // Prevent integral wind-up
120 if (integral > maxIntegral)
121 integral = maxIntegral;
122 else if (integral < -maxIntegral)
123 integral = -maxIntegral;
124
125 double derivative = error - prevError;
126
127 double output = Kp * error + Ki * integral + Kd * derivative;
128 prevError = error;
129 return output;
130 }

1017 if (targetSpeed >= 0) {
1018 // Distance from the current speed to the target speed
1019 const int speedDelta = abs(speed - targetSpeed);
1020
1021 // The time since soft movement started
1022 const long moveTimeDelta = currentTime - softStart;
1023
1024 // Calculate the time since the last PWM update
1025 const long updateTimeDelta = currentTime - lastPWMUpdate;
1026
1027 if (updateTimeDelta >= SOFT_MOVEMENT_PWM_UPDATE_INTERVAL_MICROS) {
1028 const bool timeToUpdate = moveTimeDelta < softMovingTime;
1029 // Get the true PWM update amount based on the actual elapsed time
1030 const float updateAmount =
1031 pwmUpdateAmount * (static_cast<float>(updateTimeDelta) /
1032 SOFT_MOVEMENT_PWM_UPDATE_INTERVAL_MICROS);
1033 const bool speedDeltaEnough = speedDelta >= abs(updateAmount);
1034
1035 if (timeToUpdate && speedDeltaEnough) {
1036
1037 const float newSpeed = static_cast<float>(speed + updateAmount);
1038 speed = static_cast<int>(floorf(newSpeed));
1039 // Serial.printf("Speed <== %d\n", speed);
1040 lastPWMUpdate = micros();
1041 } else {
1042 // Set the speed to the target speed and reset the soft movement if
1043 // time expired or there is less than one full update step until we
1044 // reach the target speed.
1045 speed = targetSpeed;
1046 resetSoftMovement();
1047
1048 if (requestedDirection == Direction::STOP) {
1049 systemDirection = Direction::STOP;
1050
1051 immediateHalt();
1052 stopping = false;
1053 report();
1054 }
1055 }
1056 }
1057 if (intermediateSpeed >= 0) {
1058 // Transition to intermediate speed
1059 double output = control(intermediateSpeed,
1060 speed); // Assuming speed is the current speed
1061 if (abs(output - intermediateSpeed) < 1) { // Or a small threshold
1062 intermediateSpeed = -1;
1063 }
1064 } else {
1065 // Regular target speed tracking
1066 control(targetSpeed, speed);
1067 }
1068 }

1163 }

1170 updateSoftMovement();

1173
1174 updateMotors();
1175 }
