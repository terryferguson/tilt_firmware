99 void setParams(const int kpIn, const float kiIn = DEFAULT_KI,

133 int adjustSpeed(Motor &leader, Motor &follower, const int speed,
134 const float deltaT = 0.0f) {
135
136 const float leadingMotorPosition = leader.getNormalizedPos();
137 const float laggingMotorPosition = follower.getNormalizedPos();
138 positionDifference = (laggingMotorPosition - leadingMotorPosition);
139 const float error = SETPOINT_WEIGHT * positionDifference;
140 const float derivativeTerm = K_d * filteredDerivative;
141 const float integralTerm = K_i * errorIntegral;
142 maxSpeedAdjustmentRate +=
143 MAX_ACCELERATION_INCREASE * 1000; // scaling up the increment
144 maxSpeedAdjustmentRate = std::min(
145 maxSpeedAdjustmentRate, followerMaxAccel * 1000); // scale up the limit
146
147 // Update filtered speed before it's used in PID calculations
148 filteredSpeed = SPEED_ALPHA * filteredSpeed + (1 - SPEED_ALPHA) * speed;
149
150 // Calculate the derivative and then use it to calculate the filtered
151 // derivative
152 if (deltaT > 0.0f) { // Avoid division by zero
153 derivative = (error - lastError) / deltaT;
154 filteredDerivative =
155 alpha * filteredDerivative + (1 - alpha) * derivative;
156 } else {
157 derivative = 0.0f;
158 }
159
160 const int proportionalTerm = static_cast<int>(round(error * K_p * 3));
161
162 errorIntegral += error * deltaT;
163 errorIntegral =
164 constrain(errorIntegral, limMinInteg, limMaxInteg); // Anti-Windup
165
166 const int scalarValue = leader.dir == Direction::EXTEND
167 ? POSITION_DELTA_SPEED_SCALER_EXTEND
168 : POSITION_DELTA_SPEED_SCALER_RETRACT;
169
170 const int directionSign = leader.dir == Direction::EXTEND ? 1 : -1;
171 const int positionDeltaBoost =
172 constrain(directionSign * positionDifference * scalarValue,
173 -MAX_POSITION_SCALE_VALUE, MAX_POSITION_SCALE_VALUE);
174
175 // Serial.printf("Position Delta Boost: %d\n", positionDeltaBoost);
176
177 const int u = speed - (proportionalTerm + integralTerm + derivativeTerm);
178
179 const float maxDeltaSpeed = maxSpeedAdjustmentRate * deltaT *
180 0.001; // scaling down the adjustment rate
181 int adjustedDeltaSpeed = constrain(u, -maxDeltaSpeed, maxDeltaSpeed);
182
183 int adjustedSpeed = speed + adjustedDeltaSpeed + positionDeltaBoost;
184
185 // Serial.printf("Adjusted Speed: %d\n", adjustedSpeed);
186
187 if (debugEnabled) {
188 // Debug prints
189 }
190
191 lastError = error;
192 previousMeasurement = laggingMotorPosition;
193 return constrain(adjustedSpeed, MIN_SPEED, MAX_SPEED);
194 }
